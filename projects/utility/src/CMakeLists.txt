file(GLOB_RECURSE utility_files RELATIVE
    "${CMAKE_CURRENT_SOURCE_DIR}/"

set_target_properties(utility
    PROPERTIES OUTPUT_NAME creris_utility)

    PROPERTIES VERSION ${CRERIS_VERSION} SOVERSION ${CRERIS_MAJOR_VERSION})
